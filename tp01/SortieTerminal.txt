data "aws_iam_policy_document" "force_tls" {

    sid     = "AllowSSLRequestsOnly"

      aws_s3_bucket.main.arn,
      "${aws_s3_bucket.main.arn}/*"

      type        = "*"
      identifiers = ["*"]

    condition {
      test     = "Bool"
      variable = "aws:SecureTransport"
      values   = ["false"]
    }
  }
}
